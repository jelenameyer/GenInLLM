Recalculate Reliability of all scales using Cronbach’s Alpha from psych

library(readr)
library(psych)
library(tidyverse)
library(dplyr)
library(tidyr)
items_per_LLM <- read_csv("processed_data/items_per_llm.csv", show_col_types = FALSE)
SSSV_subscale_df <- items_per_LLM %>%
  filter(category == "SSbor")
  #filter(experiment == "FTND scale")

SSSV_subscale_wide <- SSSV_subscale_df %>%
  pivot_wider(
    id_cols = c(experiment, model),
    names_from = item,
    values_from = score
  )
# Compute reliability -> with check keys (turn item orientation around again)
psych::alpha(SSSV_subscale_wide %>% select(-experiment, -model), check.keys = TRUE)
